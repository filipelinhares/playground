function checkUrl (url, cb) {
 var URL_PATTERN = /^(https?:\/\/)?([\da-z\.-]+)\.([a-z\.]{2,6})([\/\w \.-]*)*\/?$/;
  return new Promise(function (resolve, reject) {
    if (url.match(URL_PATTERN)) {
      resolve(url);
    } else {
      reject(new Error('This is not a URL'));
    }
  });
}
